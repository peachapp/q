# 代码执行结果

```js
// example 1.1
let a = {},
  b = "0",
  c = 0;
a[b] = "你好";
a[c] = "世界";
console.log(a[b]); // 世界
// => 对象和数组的区别

// example 1.2
let a = {},
  b = Symbol("1"),
  c = Symbol("1");
a[b] = "你好";
a[c] = "世界";
console.log(a[b]); // 你好
// => 自己实现Symbol

// example 1.3
let a = {},
  b = { n: "1" },
  c = { m: "2" };
a[b] = "你好";
a[c] = "世界";
console.log(a[b]); // 世界
// => Object.prototype.toString
```
